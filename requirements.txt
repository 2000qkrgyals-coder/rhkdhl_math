### 💡 기능 추가: 사진 업로드 & 시험 D-Day
오답 관리를 더 편하게 하고 싶다고 하셨죠? **"오답 번호를 적는 대신 사진을 찍어 올리는 기능"**과 **"시험 디데이"** 기능을 추가한 버전으로 코드를 업데이트해 드립니다. 

`requirements.txt`를 만드신 후, 아래 코드로 교체해 보세요.

```python
import streamlit as st
import pandas as pd
import sqlite3
from datetime import datetime
import plotly.express as px
import io

# --- [1. DB 초기화] ---
def init_db():
    conn = sqlite3.connect('tutor_smart_v1.db', check_same_thread=False)
    c = conn.cursor()
    c.execute('CREATE TABLE IF NOT EXISTS students (id INTEGER PRIMARY KEY, name TEXT, target_date TEXT)')
    c.execute('''CREATE TABLE IF NOT EXISTS sessions (
                 id INTEGER PRIMARY KEY AUTOINCREMENT, 
                 student_id INTEGER, date TEXT, session_num INTEGER,
                 curriculum TEXT, hw_score INTEGER, hw_img BLOB,
                 class_note TEXT, next_hw TEXT)''')
    conn.commit()
    return conn, c

conn, c = init_db()

# --- [2. 사이드바 & D-Day] ---
with st.sidebar:
    st.title("👨‍🏫 스마트 과외 관리")
    
    # 학생 등록
    with st.expander("➕ 학생 등록/수정"):
        s_name = st.text_input("학생 이름")
        t_date = st.date_input("다음 시험 날짜", datetime.now())
        if st.button("학생 저장"):
            c.execute("INSERT OR REPLACE INTO students (name, target_date) VALUES (?, ?)", (s_name, t_date.strftime("%Y-%m-%d")))
            conn.commit()
            st.rerun()

    students = pd.read_sql_query("SELECT * FROM students", conn)
    if not students.empty:
        sel_student = st.selectbox("학생 선택", students['name'])
        s_info = students[students['name'] == sel_student].iloc[0]
        s_id = int(s_info['id'])
        
        # D-Day 계산
        target = datetime.strptime(s_info['target_date'], "%Y-%m-%d")
        d_day = (target - datetime.now()).days
        st.metric("📅 시험 D-Day", f"D-{d_day}" if d_day >= 0 else f"D+{abs(d_day)}")
    else:
        st.stop()

# --- [3. 메인 기능 탭] ---
tab1, tab2, tab3 = st.tabs(["📝 기록하기", "📊 성취도 분석", "🖼️ 오답 갤러리"])

with tab1:
    st.subheader(f"[{sel_student}] 수업 기록")
    with st.form("session_form"):
        col1, col2 = st.columns(2)
        date = col1.date_input("날짜", datetime.now())
        curri = col2.text_input("진도/교재", placeholder="예: 미적분 쎈 1단원")
        
        st.write("📋 **숙제 성취도**")
        score = st.select_slider("점수 선택", options=[0, 20, 40, 60, 80, 100], value=80)
        
        # 핵심 기능: 오답 사진 업로드
        st.write("📸 **오답/중요 문제 사진 (폰으로 바로 촬영 가능)**")
        uploaded_file = st.file_uploader("사진을 업로드하거나 촬영하세요", type=['jpg', 'jpeg', 'png'])
        
        feedback = st.text_area("수업 피드백")
        next_hw = st.text_area("다음 숙제", value="오답 재풀이 및 연습문제 10p까지" if score < 70 else "")

        if st.form_submit_button("기록 저장"):
            img_byte = uploaded_file.getvalue() if uploaded_file else None
            c.execute("""INSERT INTO sessions (student_id, date, curriculum, hw_score, hw_img, class_note, next_hw) 
                         VALUES (?, ?, ?, ?, ?, ?, ?)""", 
                      (s_id, date.strftime("%Y-%m-%d"), curri, score, img_byte, feedback, next_hw))
            conn.commit()
            st.success("저장되었습니다!")

with tab2:
    df = pd.read_sql_query(f"SELECT * FROM sessions WHERE student_id={s_id} ORDER BY date", conn)
    if not df.empty:
        fig = px.area(df, x='date', y='hw_score', title='학습 성취도 흐름')
        st.plotly_chart(fig, use_container_width=True)
        st.dataframe(df[['date', 'curriculum', 'hw_score', 'class_note']].sort_values('date', ascending=False), hide_index=True)
    else:
        st.info("기록이 없습니다.")

with tab3:
    st.subheader("🖼️ 다시 봐야 할 오답들")
    img_df = pd.read_sql_query(f"SELECT date, curriculum, hw_img FROM sessions WHERE student_id={s_id} AND hw_img IS NOT NULL ORDER BY date DESC", conn)
    if not img_df.empty:
        for _, row in img_df.iterrows():
            with st.expander(f"📅 {row['date']} - {row['curriculum']}"):
                st.image(row['hw_img'], use_container_width=True)
    else:
        st.info("저장된 사진이 없습니다.")
